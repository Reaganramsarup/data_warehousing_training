CREATE VIEW SUPPORTING_DATA.PUBLIC.SALES_PER_LOCATION AS (
SELECT  TRUCK_ID,
        LOCATION_ID,  
        TO_DATE(ORDER_TS) AS ORDER_DATE, 
        ORDER_AMOUNT,
        SUM (ORDER_AMOUNT) OVER (PARTITION BY LOCATION_ID, TO_DATE(ORDER_TS)) AS TOTAL_PER_LOCATION,
        (ORDER_AMOUNT / SUM (ORDER_AMOUNT) OVER (PARTITION BY LOCATION_ID, TO_DATE(ORDER_TS))) * 100 AS ORDER_PERCENTAGE_OF_TOTAL
FROM HOL_DB.RAW_POS.ORDER_HEADER)
