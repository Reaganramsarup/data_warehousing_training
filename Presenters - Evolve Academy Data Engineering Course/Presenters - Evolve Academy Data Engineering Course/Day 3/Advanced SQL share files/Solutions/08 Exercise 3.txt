SELECT 
OD.ORDER_ID,
COALESCE(OD.DISCOUNT_ID, '-1') AS DISCOUNT_ID, 
COALESCE(OD.ORDER_ITEM_DISCOUNT_AMOUNT, 0) AS ORDER_ITEM_DISCOUNT_AMOUNT,
CAST(OD.ORDER_TS AS DATE) AS ORDER_DATE,
CONCAT(OH.ORDER_CURRENCY,' ',OH.ORDER_AMOUNT) AS ORDER_AMOUNT_WITH_CURR,
M.MENU_ITEM_NAME,
CONTAINS(UPPER(M.MENU_ITEM_NAME), 'WATER') AS IS_WATER,
REPLACE(UPPER(M.MENU_ITEM_NAME), 'WATER', 'H2O') AS UPDATED_NAME,
SUBSTR(M.MENU_ITEM_NAME, 1, 5) AS MENU_SHORT_NAME
FROM HOL_DB.RAW_POS.ORDER_DETAIL OD
JOIN HOL_DB.RAW_POS.ORDER_HEADER OH
ON OD.ORDER_ID = OH.ORDER_ID
JOIN HOL_DB.RAW_POS.MENU M
ON OD.MENU_ITEM_ID = M.MENU_ITEM_ID
LIMIT 100;
