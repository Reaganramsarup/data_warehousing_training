CREATE OR REPLACE PROCEDURE DATA_WAREHOUSE.PUBLIC.LOAD_DIM_DATE()
RETURNS VARCHAR NOT NULL
LANGUAGE SQL
AS
BEGIN


TRUNCATE TABLE DATA_WAREHOUSE.PUBLIC.DIM_DATE;

  INSERT INTO DATA_WAREHOUSE.PUBLIC.DIM_DATE
(
    DATE_SK,
    DATE,
    MONTH_SHORT_NAME,
    QUARTER_SHORT_NAME,
    YEAR_NUMBER,
    MONTH_NUMBER,
    DAY_NUMBER,
    PAST_YEAR,
    WEEK_NUMBER_OF_YEAR  
)

select 
    DATE_SK,
    DATE,
    MONTH_SHORT_NAME,
    QUARTER_SHORT_NAME,
    YEAR_NUMBER,
    MONTH_NUMBER,
    DAY_NUMBER,
    PAST_YEAR,
    WEEK_NUMBER_OF_YEAR
from DATA_WAREHOUSE.PUBLIC.VIRTUAL_DIM_DATE;

RETURN 'Load is complete';
END;
