
CREATE OR REPLACE VIEW DATA_WAREHOUSE.PUBLIC.VIRTUAL_DIM_FRANCHISE
AS 
SELECT 
FRANCHISE_ID,
FIRST_NAME,
LAST_NAME,
CONCAT(FIRST_NAME, ' ', LAST_NAME) AS FULL_NAME,
MAX(CITY) as CITY,
COUNTRY,
E_MAIL,
PHONE_NUMBER
FROM HOL_DB.RAW_POS.FRANCHISE
GROUP BY 
FRANCHISE_ID,
FIRST_NAME,
LAST_NAME,
COUNTRY,
E_MAIL,
PHONE_NUMBER
