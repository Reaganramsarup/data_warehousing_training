
CREATE VIEW DATA_WAREHOUSE.PUBLIC.VIRTUAL_DIM_TRUCK
AS 
SELECT 
t.TRUCK_ID,
t.PRIMARY_CITY,
t.REGION, 
t.COUNTRY,
t.YEAR,
t.MAKE, 
t.MODEL,
m.TRUCK_BRAND_NAME
FROM HOL_DB.RAW_POS.TRUCK t
LEFT JOIN 
(SELECT DISTINCT 
MENU_TYPE_ID, 
TRUCK_BRAND_NAME FROM 
HOL_DB.RAW_POS.MENU) m
on t.MENU_TYPE_ID = m.MENU_TYPE_ID
